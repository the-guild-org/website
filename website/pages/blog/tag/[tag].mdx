import { buildDynamicMeta } from '@theguild/components/compile'
import blogs from '../../../blogs.json'
import Blog from '../../blog.mdx'

export const getStaticProps = async () => {
  const props = await buildDynamicMeta()
  return { props }
}

export const getStaticPaths = async () => {
  const allTags = new Set(blogs.flatMap(art => art.tags))
  return {
    paths: Array.from(allTags).map(tag => ({ params: { tag } })),
    fallback: false
  }
}

<Blog />
